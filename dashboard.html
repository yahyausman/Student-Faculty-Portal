Database Schema

MySQL / PostgreSQL table definitions

Students Table

CREATE TABLE students (
  student_id    VARCHAR(20) PRIMARY KEY,
  first_name   VARCHAR(100) NOT NULL,
  last_name    VARCHAR(100) NOT NULL,
  email        VARCHAR(255) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  department  VARCHAR(100),
  year_of_study VARCHAR(20),
  enrollment_date DATE,
  status      ENUM('active','inactive')
    DEFAULT 'active',
  created_at  TIMESTAMP DEFAULT NOW()
);

-- Index for email lookups
CREATE INDEX idx_email
  ON students(email);

Faculty Table

CREATE TABLE faculty (
  employee_id  VARCHAR(20) PRIMARY KEY,
  first_name   VARCHAR(100) NOT NULL,
  last_name    VARCHAR(100) NOT NULL,
  email        VARCHAR(255) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  department  VARCHAR(100) NOT NULL,
  title        VARCHAR(60),
  specialization VARCHAR(255),
  hire_date   DATE,
  status      ENUM('active','inactive')
    DEFAULT 'active',
  created_at  TIMESTAMP DEFAULT NOW()
);

Relational Tables

Courses, enrollments, and grades

CREATE TABLE courses (
  course_id    VARCHAR(20) PRIMARY KEY,
  course_name  VARCHAR(200) NOT NULL,
  credits      INT,
  instructor_id VARCHAR(20),
  FOREIGN KEY(instructor_id) REFERENCES faculty(employee_id)
);

CREATE TABLE enrollments (
  id           INT AUTO_INCREMENT PRIMARY KEY,
  student_id   VARCHAR(20),
  course_id    VARCHAR(20),
  semester     VARCHAR(20),
  grade        VARCHAR(5),
  FOREIGN KEY(student_id) REFERENCES students(student_id),
  FOREIGN KEY(course_id)  REFERENCES courses(course_id)
);

Flask Route Examples

How pages and forms connect to the database

# app.py — Python Flask + MySQL (no REST needed)
from flask import Flask, render_template, request, redirect, session
import mysql.connector

app = Flask(__name__)
app.secret_key = 'your_secret_key'

# Connect to MySQL
def get_db():
  return mysql.connector.connect(
    host="localhost", user="root",
    password="yourpassword", database="academia_db"
  )

# Student Registration
@app.route('/register/student', methods=['GET','POST'])
def register_student():
  if request.method == 'POST':
    db = get_db()
    cursor = db.cursor()
    cursor.execute("INSERT INTO students (student_id, first_name, last_name, email, department) VALUES (%s,%s,%s,%s,%s)",
      (request.form['student_id'], request.form['first_name'],
       request.form['last_name'], request.form['email'],
       request.form['department']))
    db.commit()
    return redirect('/dashboard')
  return render_template('register_student.html')

# Student Login
@app.route('/login/student', methods=['GET','POST'])
def login_student():
  if request.method == 'POST':
    db = get_db()
    cursor = db.cursor(dictionary=True)
    cursor.execute("SELECT * FROM students WHERE student_id=%s",
      (request.form['student_id'],))
    student = cursor.fetchone()
    if student:
      session['user'] = student
      return redirect('/dashboard')
  return render_template('login.html')

# Dashboard — render page with DB data
@app.route('/dashboard')
def dashboard():
  student = session.get('user')
  return render_template('dashboard.html', student=student)
